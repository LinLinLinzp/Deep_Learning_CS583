class Worker:
  def __init__(self,x,y):
    self.x = x 
    self.y = y
    self.s = x.shape[0]
    self.d = x.shape[1]
    self.w = np.zeros((d,1))

  # set the model para to the latest
  def set_param(self, w):
    self.w = w

  # compute the local loss
  def loss(self):
    yx = np.multiply(self.x,self.y) #s-d
    yxw = np.dot(yx,self.w) # s-1
    vec1 = np.exp( -yxw) #s-1
    vec2 = np.log(1 + vec1) #s-1
    return np.sum(vec2)

  # compute local grad
  def gradient(self):
    yx = np.multiply(self.y,self.x) # s-d
    yxw = np.dot(yx,self.w) # s-1
    vec1 = np.exp(yxw) #s-1
    vec2 = np.divide(yx, 1+vec1) # s-d
    g = -np.sum(vec2 , axis=0).reshape(self.d,1) #d-1
    return g

  # gd
  def gd(self,q, lr):
    grads = 0
    for i in range(q):
      grad = self.gradient()
      self.w -= lr * grad
      grads = grad
    return grads

  # sgd grad
  def sgd_gradient(self,w, xi, yi, lam):
    d = xi.shape[0]
    yx = yi * xi # 1-by-d matrix
    yxw = float(numpy.dot(yx, w)) # scalar
    
    # calculate objective function Q_i
    loss = numpy.log(1 + numpy.exp(-yxw)) # scalar
    reg = lam / 2 * numpy.sum(w * w) # scalar
    obj = loss + reg
    
    # calculate stochastic gradient
    g_loss = -yx.T / (1 + numpy.exp(yxw)) # d-by-1 matrix
    g = g_loss + lam * w # d-by-1 matrix
    
    return obj, g

  # sgd
  def sgd(self, q, lr, lam):
    grads = 0
    for i in range(q):
      # randomly shuffle the samples
      rand_indices = numpy.random.permutation(self.s)
      x_rand = self.x[rand_indices, :]
      y_rand = self.y[rand_indices, :]
        
      objval = 0 # accumulate the objective values
      for i in range(self.s):
        xi = x_rand[i, :] # 1-by-d matrix
        yi = float(y_rand[i, :]) # scalar
        obj, g = self.sgd_gradient(w, xi, yi, lam)
        objval += obj
        self.w -= lr * g
        grads = g
    return grads

class Server:
  def __init__(self,m,n,d):
    self.m = m # work nodes
    self.n = n # training samples
    self.d = d # features
    self.w = np.zeros((d,1)) # param
    self.g = np.zeros((d,1)) # grad
    self.v = np.zeros((d,1)) # momentum
    self.loss = 0
    self.obj = 0

  def broadcast(self):
    return self.w

  # sum the grad and loss from workers
  def aggregate(self, grads, losses):
    self.g = np.zeros((self.d,1))
    self.loss = 0
    for k in range(self.m):
      self.g += grads[k]
      self.loss += losses[k]
  
  # compute the grad (from loss and regular)
  def gradient(self, lam):
    self.g = self.g / self.n + lam * self.w
  
  # compute the obj function
  def objective(self, lam):
    reg= lam / 2 * np.sum(self.w * self.w)
    self.obj = self.loss  / self.n + reg
    return self.obj

  # update the param using agd
  # alpha: lr
  # beta: momentum param
  def agd(self, alpha, bate):
    self.v *= beta
    self.v += self.g
    self.w -= alpha * self.v

import math

# create a server and m work nodes
def create_server_workers(m,x,y):
  n, d = x.shape
  s = math.floor(n / m)
  server = Server(m,n,d)
  workers = []

  for i in range(m):
    indices = list(range(i * s, (i+1)*s))
    worker = Worker(x[indices,:],y[indices,:])
    workers.append(worker)
  
  return server, workers
